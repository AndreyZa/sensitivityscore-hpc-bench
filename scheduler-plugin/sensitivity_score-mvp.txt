package sensitivityscore

import (
	"context"
	"encoding/json"
	"os"
	"sync"
	"time"

	v1 "k8s.io/api/core/v1"
	"k8s.io/apimachinery/pkg/runtime"
	"k8s.io/klog/v2"
	framework "k8s.io/kube-scheduler/framework"
)

const Name = "SensitivityScore"

// sensitivityAnnotation - аннотация на поде, помечающая задачу как
// чувствительную к интерференции на LLC (Last Level Cache).
// Пример использования в манифесте пода:
//
//	metadata:
//	  annotations:
//	    sensitivity.scheduling.x-k8s.io/llc-sensitive: "true"
const sensitivityAnnotation = "sensitivity.scheduling.x-k8s.io/llc-sensitive"

// metricsFilePath - путь к файлу с метриками загрузки нод.
// Монтируется через ConfigMap (см. ниже).
const metricsFilePath = "/etc/sensitivity/node-metrics.json"

// refreshInterval - как часто перечитывать файл с метриками.
const refreshInterval = 10 * time.Second

// nodeMetrics - формат записи в JSON-файле: имя ноды -> "уровень шума" 0-100,
// где 100 означает "на ноде уже работает интенсивная memory-bound задача,
// размещать сюда LLC-чувствительные задачи нежелательно".
type nodeMetrics map[string]float64

// SensitivityScore - плагин с in-memory кэшем метрик.
type SensitivityScore struct {
	handle framework.Handle

	mu      sync.RWMutex
	metrics nodeMetrics
}

var _ framework.ScorePlugin = &SensitivityScore{}

func (s *SensitivityScore) Name() string {
	return Name
}

// Score - читает текущий "шум" на ноде из кэша (без сетевых вызовов) и
// сочетает его с чувствительностью пода.
func (s *SensitivityScore) Score(
	ctx context.Context,
	state framework.CycleState,
	pod *v1.Pod,
	nodeInfo framework.NodeInfo,
) (int64, *framework.Status) {
	nodeName := nodeInfo.Node().Name

	s.mu.RLock()
	noise := s.metrics[nodeName] // 0, если для ноды нет данных
	s.mu.RUnlock()

	sensitive := pod.Annotations[sensitivityAnnotation] == "true"

	var score int64
	if sensitive {
		// Чувствительный под: чем выше "шум" на ноде, тем ниже score.
		// noise=0   -> score=100 (отличный выбор)
		// noise=100 -> score=0   (худший выбор)
		score = int64(100 - noise)
	} else {
		// Нечувствительный под: уровень шума не важен, нейтральная оценка.
		// Такие поды можно безопасно класть на "шумные" ноды.
		score = 50
	}

	if score < framework.MinNodeScore {
		score = framework.MinNodeScore
	}
	if score > framework.MaxNodeScore {
		score = framework.MaxNodeScore
	}

	klog.InfoS("SensitivityScore.Score called",
		"pod", pod.Name, "node", nodeName,
		"sensitive", sensitive, "noise", noise, "score", score)

	return score, nil
}

func (s *SensitivityScore) ScoreExtensions() framework.ScoreExtensions {
	return nil
}

// New - конструктор. Запускает фоновую горутину обновления кэша,
// привязанную к контексту жизни scheduler'а: когда scheduler завершается,
// ctx закрывается и горутина останавливается сама.
func New(ctx context.Context, _ runtime.Object, h framework.Handle) (framework.Plugin, error) {
	s := &SensitivityScore{
		handle:  h,
		metrics: make(nodeMetrics),
	}

	go s.refreshLoop(ctx)

	return s, nil
}

// refreshLoop - раз в refreshInterval перечитывает файл с метриками.
func (s *SensitivityScore) refreshLoop(ctx context.Context) {
	s.reload()

	ticker := time.NewTicker(refreshInterval)
	defer ticker.Stop()

	for {
		select {
		case <-ctx.Done():
			return
		case <-ticker.C:
			s.reload()
		}
	}
}

// reload - читает и парсит JSON-файл, атомарно заменяет карту метрик.
func (s *SensitivityScore) reload() {
	data, err := os.ReadFile(metricsFilePath)
	if err != nil {
		klog.ErrorS(err, "failed to read sensitivity metrics file", "path", metricsFilePath)
		return
	}

	var m nodeMetrics
	if err := json.Unmarshal(data, &m); err != nil {
		klog.ErrorS(err, "failed to parse sensitivity metrics file")
		return
	}

	s.mu.Lock()
	s.metrics = m
	s.mu.Unlock()

	// klog.InfoS("sensitivity metrics reloaded", "nodes", len(m))
}
